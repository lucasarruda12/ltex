\def\pdfcolorstackinit {\pdffeedback colorstackinit}
\protected\def\pdfcolorstack {\pdfextension colorstack}

\newcount\colorRegCount
\colorRegCount=\pdfcolorstackinit {0 g 0 G}

\def\newColor #1 #2 %
{\expandafter\def\csname text#1\endcsname %
		{\pdfcolorstack \colorRegCount set {#2 rg}}%
\expandafter\def\csname #1\endcsname##1 %
		{\pdfcolorstack \colorRegCount push {#2 rg}##1\pdfcolorstack \colorRegCount pop {}}}
